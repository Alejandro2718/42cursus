char	*get_next_line(int fd)
{
	char			*line;
	static char		*buffer = NULL;
	static int		position = 0;
	static size_t	bytes_read = 0;
	int				i;

	if (position == 0)
	{
		buffer = malloc(BUFFER_SIZE + 1);
		if (!buffer)
			return (NULL);
		bytes_read = read(fd, buffer, BUFFER_SIZE);
		if (bytes_read == (size_t)(-1))
		{
			free(buffer);
			return (NULL);
		}
		buffer[bytes_read] = '\0';
	}
	line = malloc(bytes_read + 1);
	if (!line)
	{
		free(buffer);
		free(line);
		buffer = NULL;
		position = 0;
		return (NULL);
	}
	i = 0;
	while ((size_t)position <= bytes_read + 1 && buffer[position] != '\0')
	{
		if (buffer[position] == '\n' || buffer[position] == '\0')
		{
			if (buffer[position] == '\n')
			{
				line[i] = '\n';
				line[i + 1] = '\0';
			}
			else
			{
				line[i] = '\0';
			}
			position++;
			return (line);
		}
		line[i] = buffer[position];
		i++;
		position++;
	}
	free(buffer);
	free(line);
	return (NULL);
}
